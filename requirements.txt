import streamlit as st  
import pandas as pd  
  
# 设置页面标题  
st.title('FAQ 上传与查看')  
  
# 上传Excel文件  
uploaded_file = st.file_uploader("请上传FAQ Excel文件", type=["xlsx"])  
  
if uploaded_file is not None:  
    # 读取Excel文件  
    df = pd.read_excel(uploaded_file)  
      
    # 显示数据框  
    st.write("上传的FAQ内容：")  
    st.dataframe(df)  
else:  
    st.write("请上传一个Excel文件以查看FAQ内容。")
